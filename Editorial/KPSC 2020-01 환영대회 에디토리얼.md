#include <iostream>
#include <vector>
#include <set>
#include <queue>
#include <string>
#define MAXV 100000
#define MAXL 100000
using namespace std;

struct strt{
    int v, t;
};
int N, L, S, T, ans = -1;
vector<int> Lines[MAXL + 1], Stas[MAXV + 1];
set<int> Lines_L[MAXL + 1];
bool V[MAXL + 1], VS[MAXV + 1], C[MAXL + 1];
int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> N >> L;
    for(int j = 1; j <= L; j++){
        int k; cin >> k;
        while(k--){
            int v; cin >> v;
            Lines[j].push_back(v);
            Stas[v].push_back(j);
        }
    }
    cin >> S >> T;
    queue<strt> q;
    for(int j = 0; j < Stas[S].size(); j++){
        q.push({Stas[S][j], 0});
        V[Stas[S][j]] = true;
    }
    for(int j = 0; j < Stas[T].size(); j++){
        C[Stas[T][j]] = true;
    }
    while(!q.empty()){
        strt t = q.front(); q.pop();

        if(C[t.v]){
            ans = t.t;
            break;
        }

        for(auto itr = Lines[t.v].begin(); itr != Lines[t.v].end(); itr++){
            int u = (*itr);
            if(VS[u])
                continue;
            VS[u] = true;

            for(int j = 0; j < Stas[u].size(); j++){
                if(V[Stas[u][j]])
                    continue;
                V[Stas[u][j]] = true;
                q.push({Stas[u][j], t.t + 1});
            }
        }
    }
    cout << ans;
    return 0;
}

## D. 적절한수열

* 출제자: 박정현



이분탐색

수열을 이루는 수가 모두 양수이기때문에 누적합은 항상 증가합니다.

구간의 시작점을 차례대로 고정시켜보면서 구간의 합이 K이상이 되는 첫 번째 점을 이분탐색으로 찾을 수 있습니다.

찾은 점부터 수열의 끝점까지는 항상 K이상의 합이 보장되기때문에 모든 시작점에 대해 구한 개수를 모두 합하면 됩니다.

시간복잡도는 O(N Log N), 다른 풀이로는 투 포인터 알고리즘으로 O(N)만에 구해낼 수도 있습니다.



```c++
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <climits>
#include <set>
#include <map>
#include <queue>
#include <deque>
#include <stack>
#include <string>
#include <list>
#include <ctime>
#include <complex>
#include <bitset>

#define ff first
#define ss second

using namespace std;
using pii=pair<int,int>;
using ll=long long;

int main()
{
    ios::sync_with_stdio(false);
	cin.tie(0);

    ll n,k;
    cin>>n>>k;
    vector<ll> v(n);
    for(int i=0; i<n; i++){
        cin>>v[i];
        if(i>0) v[i]+=v[i-1];
    }
    ll answer=0;
    for(int i=0; i<n; i++){
        int lo=i,hi=n;
        while(lo<hi){
            int mid=(lo+hi)/2;
            ll sum=v[mid];
            if(i>0) sum-=v[i-1];
            if(sum<k) lo=mid+1;
            else hi=mid;
        }
        answer+=n-lo;
    }
    cout<<answer;

    return 0;
}
```

int N, U[MAXN], cnt[MAXN + 1], ans;

    ios::sync_with_stdio(false);

        cin >> U[j];
    for(int j = 0; j < N; j++){
        cnt[U[j]]++;
        if(cnt[U[j]] - 1 > 0)
            mset.erase(mset.find(cnt[U[j]] - 1));
        mset.insert(cnt[U[j]]);
        if(mset.size() == 1){
            ans = j + 1;
            continue;
        }
        auto itr1 = mset.begin();
        auto itr2 = mset.end();
        if((*itr1) - 1  == 0){
            itr1++; itr2--;
            if((*itr1) == (*itr2)){
                ans = j + 1;
                continue;
            }
        }
        itr1 = mset.begin();
        itr2 = mset.end();
        itr2--;
        if((*itr2) - 1 == (*itr1)){
            itr2--;
            if((*itr1) == (*itr2)){
                ans = j + 1;
            }
        }
    }
    cout << ans;
    return 0;
}

                        nab2 += P[c].b;
                        nsqsum2 += P[c].r * P[c].r;
                        nsqsum2 += P[c].g * P[c].g;
                        nsqsum2 += P[c].b * P[c].b;
                        nsum2r += P[c].r;
                        nsum2g += P[c].g;
                        nsum2b += P[c].b;
                        ncnt2++;
                    }

                    nar1 /= ncnt1; nag1 /= ncnt1; nab1 /= ncnt1;
                    nar2 /= ncnt2; nag2 /= ncnt2; nab2 /= ncnt2;

                    c1 = nsqsum1 - 2 * (nsum1r * nar1 + nsum1g * nag1 + nsum1b * nab1) + ncnt1 * (nar1 * nar1 + nag1 * nag1 + nab1 * nab1);
                    c2 = nsqsum2 - 2 * (nsum2r * nar2 + nsum2g * nag2 + nsum2b * nab2) + ncnt2 * (nar2 * nar2 + nag2 * nag2 + nab2 * nab2);

                    ans = min(ans, c1 + c2);
                }

            }
        }
    }
    cout << fixed << ans;
    return 0;
}
